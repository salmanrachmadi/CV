\section{Pendahuluan}

\begin{frame}{Latar Belakang \& Masalah}
\begin{columns}
\begin{column}{0.6\textwidth}
\textbf{Pentingnya Deteksi Helm:}
\begin{itemize}
    \item Helm merupakan pelindung utama pengendara motor dari cedera kepala fatal dalam kecelakaan lalu lintas.
    \item Pemantauan kepatuhan secara manual tidak efisien dan tidak dapat diskalakan untuk volume lalu lintas nyata.
    \item Diperlukan otomatisasi berbasis \textbf{Computer Vision} (deteksi objek) dari kamera CCTV jalan raya.
\end{itemize}

\vspace{0.5em}
\textbf{Dua Asumsi yang Sering Tak Diuji:}
\begin{itemize}
    \item Bahwa bobot pra-latih COCO bisa langsung dipakai pada domain helm.
    \item Bahwa menambah mekanisme atensi (mis. CBAM) otomatis meningkatkan akurasi, padahal klaim ini sering tanpa uji signifikansi \& multi-seed.
\end{itemize}
\end{column}
\begin{column}{0.4\textwidth}
\centering
\begin{tikzpicture}[scale=0.8]
    \draw[thick, rounded corners, fill=yoloblue!10] (0,3) rectangle (4,4.5) node[midway, align=center] {\small Pemantauan\\Manual};
    \draw[->, thick, accent] (2,3) -- (2,2) node[midway, right] {\small Lambat, Mahal};
    \draw[thick, rounded corners, fill=success!10] (0,0.5) rectangle (4,2) node[midway, align=center] {\small Deteksi Otomatis\\(YOLO + Atensi?)};
\end{tikzpicture}
\end{column}
\end{columns}
\end{frame}

\begin{frame}{Pertanyaan Riset}
\textbf{Kontribusi utama bersifat metodologis:} memisahkan secara eksplisit efek \emph{fine-tuning} (gap domain) dari efek \emph{atensi}, dengan protokol identik, multi-seed, dan uji statistik.

\vspace{1em}
\begin{block}{RQ1 --- Gap Domain}
Seberapa besar gap domain antara bobot COCO apa adanya dan tugas deteksi helm?
\end{block}

\begin{block}{RQ2 --- Peran Atensi}
Setelah fine-tuning dengan protokol identik dan multi-seed, apakah mekanisme atensi (C2PSA pada YOLO11s, CBAM pada YOLOv8s) melampaui CNN murni YOLOv8s secara \textbf{statistik bermakna}?
\end{block}

\vspace{0.5em}
\footnotesize Catatan: tujuannya bukan menyimpulkan ``atensi buruk'', melainkan menguji asumsi secara terkendali.
\end{frame}

\section{Metodologi}

\begin{frame}{Dataset \& Kelas Target}
\begin{columns}
\begin{column}{0.55\textwidth}
\textbf{Karakteristik Dataset:}
\begin{itemize}
    \item Dataset publik Roboflow Universe: \textbf{NCKH 2023 Helmet Detection Project (v19)}, lisensi MIT.
    \item Total \textbf{1.803 gambar}, split tetap:
    \begin{itemize}
        \item \textbf{Train:} 1.563 \quad \textbf{Val:} 140 \quad \textbf{Test:} 100
    \end{itemize}
    \item Resolusi input \textbf{1280 piksel} untuk mempertahankan detail objek kecil.
    \item Kondisi relatif mudah (siang hari, pencahayaan memadai, oklusi ringan).
\end{itemize}
\end{column}
\begin{column}{0.45\textwidth}
\textbf{3 Kelas Target Deteksi:}
\begin{enumerate}
    \item \textbf{\texttt{motorcyclist}:} pengendara (konteks utama).
    \item \textbf{\texttt{helmet}:} helm (objek kepatuhan).
    \item \textbf{\texttt{license\_plate}:} plat nomor (jangkar identifikasi).
\end{enumerate}
\end{column}
\end{columns}
\end{frame}

\begin{frame}{Arsitektur Model yang Dibandingkan}
\centering
\begin{tabular}{llcc}
\toprule
\textbf{Model} & \textbf{Jenis Atensi} & \textbf{Jumlah Parameter} & \textbf{Peran Eksperimen} \\
\midrule
\textbf{YOLOv8s} & None & $\sim$11,17 Juta & Baseline (Pure CNN) \\
\textbf{YOLO11s} & Partial (C2PSA) & $\sim$9,40 Juta & Atensi parsial (self-attention) \\
\textbf{YOLOv8s+CBAM} & Channel + Spatial & $\sim$11,51 Juta & Intervensi atensi eksplisit \\
\bottomrule
\end{tabular}

\vspace{1.5em}
\begin{columns}
\begin{column}{0.9\textwidth}
\begin{itemize}
    \item \textbf{YOLOv8s+CBAM}: 3 modul CBAM disisipkan tepat sebelum kepala deteksi (P3, P4, P5); parameter naik $\sim 3\%$.
    \item CBAM diuji pada \textbf{satu konfigurasi saja} (penempatan P3/P4/P5, \textit{learning rate} tunggal, tanpa pemanasan terpisah) --- keterbatasan yang diakui.
\end{itemize}
\end{column}
\end{columns}
\end{frame}

\begin{frame}{Dua Protokol Evaluasi}
\begin{block}{Protokol 1 --- Baseline Vanilla (Zero-shot COCO)}
Bobot COCO YOLOv8s/YOLO11s dievaluasi \textbf{tanpa pelatihan}. \texttt{motorcycle} COCO dipetakan ke \texttt{motorcyclist}; \texttt{helmet}/\texttt{license\_plate} tak ada di COCO sehingga AP-nya \textbf{nol \emph{by design}}. Dihitung dengan \texttt{pycocotools}. Tujuannya memverifikasi gap domain, bukan membandingkan arsitektur.
\end{block}

\begin{block}{Protokol 2 --- Fine-tuning (Transfer Learning)}
Ketiga model dilatih dari bobot COCO dengan protokol \textbf{identik}: 1280 px, 100 epoch, optimizer/augmentasi sama (mosaic, HSV, flip, scale), early stopping (patience 25). \textbf{Multi-seed} (42, 0, 1) dijalankan \textbf{berurutan \& mandiri} dengan pembersihan memori GPU antar-run.
\end{block}

\vspace{0.3em}
\footnotesize \textbf{Uji statistik:} paired t-test antar-seed pada \emph{kedua} metrik (mAP@0.5 \& mAP@[.5:.95]) + Cohen's $d$, $\alpha = 0,05$. Dengan $n=3$, daya uji terbatas.
\end{frame}

\section{Pembahasan}

\begin{frame}{Mengapa Atensi Tidak Terbayar di Sini?}
\begin{enumerate}
    \item \textbf{Efek langit-langit (ceiling effect):}
    \begin{itemize}
        \item Baseline sudah $\sim$0,96 pada dataset relatif mudah --- ruang perbaikan sangat tipis bagi atensi.
        \item Manfaat atensi umumnya muncul pada data besar \& beragam.
    \end{itemize}
    \vspace{0.4em}
    \item \textbf{Disrupsi inisialisasi CBAM:}
    \begin{itemize}
        \item Modul CBAM diinisialisasi acak, sisanya mewarisi bobot COCO; dengan data terbatas \& \textit{learning rate} tunggal, fitur matang terganggu.
    \end{itemize}
    \vspace{0.4em}
    \item \textbf{Hasil CBAM bukan vonis umum:}
    \begin{itemize}
        \item Terbatas pada satu konfigurasi (penempatan/LR/warmup). Konfigurasi lain bisa berbeda --- perlu ablasi sistematis.
    \end{itemize}
    \vspace{0.4em}
    \item \textbf{Atensi parsial (C2PSA) lebih ``netral''} daripada modul tempelan pasca-hoc (CBAM), namun tetap tanpa keunggulan meyakinkan.
\end{enumerate}
\end{frame}

\begin{frame}{Kesimpulan \& Saran}
\begin{columns}
\begin{column}{0.5\textwidth}
\textbf{Kesimpulan Utama:}
\begin{itemize}
    \item \textbf{Gap domain besar} ($\sim$0,10 zero-shot $\rightarrow$ $\sim$0,96 fine-tuned): \textbf{fine-tuning wajib}.
    \item \textbf{Tidak ada atensi yang konsisten melampaui CNN murni}: YOLOv8s unggul di mAP@0.5; YOLO11s sedikit unggul di mAP@[.5:.95] (tidak signifikan).
    \item CBAM menurunkan mAP@0.5 secara signifikan, tetapi pada \textbf{satu konfigurasi}.
\end{itemize}
\end{column}
\begin{column}{0.5\textwidth}
\textbf{Saran Riset Lanjut:}
\begin{itemize}
    \item Dataset lebih menantang (malam, hujan, oklusi) yang mungkin memberi ruang bagi atensi.
    \item \textbf{Ablasi CBAM} sistematis: penempatan di backbone, \textit{from scratch}, LR terpisah.
    \item Tambah seed ($n \ge 5$) + analisis daya \emph{a priori} untuk memperkuat klaim.
    \item Catatan etika/privasi pada integrasi deteksi helm + plat nomor.
\end{itemize}
\end{column}
\end{columns}
\end{frame}
