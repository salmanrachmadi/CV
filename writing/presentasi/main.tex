\section{Metodologi}

\begin{frame}{Dataset \& Kelas Sasaran}
\begin{columns}
\begin{column}{0.5\textwidth}
    \large\textbf{Dataset: Roboflow NCKH}
    \normalsize
    \vspace{0.5em}
    \begin{itemize}
        \item \textbf{1.803 Gambar} (1.563 / 140 / 100 = Train/Val/Test) $\approx$ 8.229 instance.
        \item \textbf{Augmentasi ekspor:} rotasi $\pm5^\circ$, shear $\pm5^\circ$, kecerahan $\pm20\%$, salt-and-pepper 5\%.
        \item \textbf{Resolusi 1280px}: menangkap objek kecil (helm \& plat) di kejauhan.
        \item Kondisi jalan raya luar ruangan yang realistis.
    \end{itemize}
\end{column}

\begin{column}{0.5\textwidth}
    \large\textbf{3 Kelas Target Deteksi}
    \normalsize
    \vspace{0.5em}
    \begin{itemize}
        \item \texttt{license\_plate}: 42,5\% (terbanyak).
        \item \texttt{motorcyclist}: 33,5\%.
        \item \texttt{helmet}: 24,0\% (paling sedikit \& terkecil $\Rightarrow$ tersulit).
    \end{itemize}
    \vspace{0.5em}
    \footnotesize \textit{*Note:} Total 8.229 instance; ketidakseimbangan kelas alami.
\end{column}
\end{columns}
\end{frame}

\section{Hasil Eksperimen}

\begin{frame}{Hasil Akurasi Setelah Fine-Tuning}
\begin{columns}
\begin{column}{0.52\textwidth}
    \centering
    \small
    \resizebox{\columnwidth}{!}{%
    \begin{tabular}{lccc}
    \toprule
    \textbf{Model} & \textbf{mAP@0.5} & \textbf{mAP@[.5:.95]} & \textbf{FPS} \\
    \midrule
    \textbf{YOLOv8s} & $\mathbf{0,9596 \pm 0,0008}$ & $0,6885 \pm 0,0006$ & $\mathbf{\sim 296}$ \\
    YOLO11s          & $0,9574 \pm 0,0019$          & $\mathbf{0,6902 \pm 0,0022}$ & $\sim 189$ \\
    YOLOv8s+CBAM     & $0,9481 \pm 0,0006$          & $0,6832 \pm 0,0034$ & $\sim 290$ \\
    \bottomrule
    \end{tabular}%
    }
    
    \vspace{1.2em}
    \normalsize
    \begin{itemize}
        \item \textbf{Akurasi Kasaran (mAP@0.5)}: YOLOv8s murni memimpin dan paling cepat.
        \item \textbf{Presisi Lokalisasi (mAP@[.5:.95])}: YOLO11s (Attention) unggul tipis ($+0,0017$), namun lambat.
        \item \textbf{CBAM (Attention tambahan)}: Justru mengalami penurunan performa.
    \end{itemize}
\end{column}

\begin{frame}{Uji Signifikansi Statistik (Paired t-test)}
    \begin{columns}[t]
        \begin{column}{0.48\textwidth}
            \centering
            \colorbox{accent!10}{\makebox[0.95\textwidth]{\strut\textbf{\color{accent}Attention Tambahan (CBAM) Gagal}}}
            \vspace{0.8em}
            
            \small
            \begin{itemize}
                \item \textbf{Turun Nyata}: Akurasi mAP@0.5 turun signifikan dibandingkan baseline ($p = 0,004$).
                \item Efek ukuran sangat besar (Cohen's $d = 9,3$).
            \end{itemize}
        \end{column}
        
        \begin{column}{0.48\textwidth}
            \centering
            \colorbox{primary!10}{\makebox[0.95\textwidth]{\strut\textbf{\color{primary}YOLO11s vs YOLOv8s}}}
            \vspace{0.8em}
            
            \small
            \begin{itemize}
                \item \textbf{Tidak Cukup Bukti Perbedaan}: Perbedaan tidak signifikan secara statistik ($p = 0,11$ untuk mAP@0.5, $p = 0,37$ untuk mAP@[.5:.95]).
                \item \textbf{Daya Statistik Rendah ($n=3$)}: Klaim kesetaraan definitif belum dapat diambil (*absence of evidence*).
            \end{itemize}
        \end{column}
    \end{columns}
    
    \vspace{2.2em}
    \centering
    \large\textbf{\color{secondary}Jawaban RQ2: Attention TIDAK memberikan keuntungan nyata.}
\end{frame}

\section{Pembahasan}

\begin{frame}{Diskusi Temuan: Mengapa Attention Kurang Efektif?}
    \begin{columns}[t]
        \begin{column}{0.48\textwidth}
            \centering
            \colorbox{primary!10}{\makebox[0.95\textwidth]{\strut\textbf{\color{primary}1. Efek Langit-langit (Ceiling)}}}
            \vspace{0.8em}
            
            \small
            Akurasi model baseline CNN murni (YOLOv8s) sudah terlampau tinggi (\textbf{95,9\%}). Ruang peningkatan untuk modul attention sangat sempit pada dataset ini.
        \end{column}
        
        \begin{column}{0.48\textwidth}
            \centering
            \colorbox{primary!10}{\makebox[0.95\textwidth]{\strut\textbf{\color{primary}2. Gangguan Fitur (CBAM)}}}
            \vspace{0.8em}
            
            \small
            Inisialisasi acak modul CBAM pada model pretrained COCO mengganggu representasi fitur yang sudah matang, merusak akurasi tanpa proses pemanasan (\textit{warmup}) yang lama.
        \end{column}
    \end{columns}
    
    \vspace{1.5em}
    \centering
    \footnotesize \textit{*Catatan Keterbatasan:} Temuan negatif CBAM terbatas pada konfigurasi uji (P3/P4/P5, tanpa warmup terpisah, single learning rate) dan tidak berlaku umum untuk seluruh konfigurasi CBAM.
\end{frame}
